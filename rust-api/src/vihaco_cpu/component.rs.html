11impl Reset for CPU {
12    fn reset(&mut self) {
13        self.frames.clear();
14        self.heap.clear();
15        self.stack.clear();
16        self.span = (0, 0, 0);
17        self.pending_pc = None;
18        self.current_pc = 0;
19        self.return_values.clear();
20    }
21}
22
23impl CPU {
24    pub fn execute_instruction(&mut self, inst: Instruction) -> eyre::Result<StepOutcome> {
25        self.clear_pending_pc();
26        use Instruction::*;
27        match inst {
28            Span(file, start, end) => self.op_span(file, start, end),
29            Label | FunctionStart | FunctionEnd => Ok(StepOutcome::Continue),
30            Breakpoint => Ok(StepOutcome::Breakpoint),
31            Branch(target) => self.op_branch(target),
32            ConditionalBranch(true_target, false_target) => {
33                self.op_conditional_branch(true_target, false_target)
34            }
35            Return(keep) => self.op_return(keep),
36            Call(arity, target) => self.op_call(arity, target),
37            IndirectCall => self.op_indirect_call(),
38            Halt => Ok(StepOutcome::Halt),
39            Print => Err(eyre::eyre!(
40                "Print must be handled via execute with CPUMessage::Print"
41            )),
42            Load(ty, addr) => self.op_load(ty, addr),
43            Store(ty, addr) => self.op_store(ty, addr),
44            Dup => self.op_dup(),
45            HeapAlloc(n_elements) => self.op_heap_alloc(n_elements),
46            GetItem => self.op_get_item(),
47            HeapDealloc => self.op_heap_dealloc(),
48            Const(v) => self.op_const(v),
49            Add(ty) => self.op_add(ty),
50            Sub(ty) => self.op_sub(ty),
51            Mul(ty) => self.op_mul(ty),
52            Div(ty) => self.op_div(ty),
53            Rem(ty) => self.op_rem(ty),
54            Neg(ty) => self.op_neg(ty),
55            Shl(ty) => self.op_shl(ty),
56            Shr(ty) => self.op_shr(ty),
57            Rol(ty) => self.op_rol(ty),
58            Ror(ty) => self.op_ror(ty),
59            BitAnd(ty) => self.op_bitand(ty),
60            BitOr(ty) => self.op_bitor(ty),
61            BitXor(ty) => self.op_bitxor(ty),
62            Not => self.op_not(),
63            And => self.op_and(),
64            Or => self.op_or(),
65            Xor => self.op_xor(),
66            Eq(ty) => self.op_eq(ty),
67            Ne(ty) => self.op_ne(ty),
68            Lt(ty) => self.op_lt(ty),
69            Gt(ty) => self.op_gt(ty),
70            Le(ty) => self.op_le(ty),
71            Ge(ty) => self.op_ge(ty),
72        }
73    }
74}

76#[derive(Debug, Clone, PartialEq, vihaco::Message)]
77pub enum CPUMessage {
78    None,
79    FunctionInfo { arity: u32, start_address: u32 },
80    Print(String),
81}
82
83#[component(instruction = Instruction, message = CPUMessage, effect = StepOutcome)]
84impl CPU {
85    fn execute(
86        &mut self,
87        inst: Instruction,
88        msg: CPUMessage,
89    ) -> eyre::Result<Effects<StepOutcome>> {
90        use Instruction::*;
91        match (inst, msg) {
92            (Print, CPUMessage::Print(text)) => {
93                self.stack_pop()?;
94                drop(text);
95                Ok(Effects::one(StepOutcome::Continue))
96            }
97            (Print, _) => Err(eyre::eyre!("Print requires CPUMessage::Print")),
98            (_, CPUMessage::Print(_)) => Err(eyre::eyre!(
99                "CPUMessage::Print is only valid for Print instruction"
100            )),
101            (
102                inst,
103                CPUMessage::FunctionInfo {
104                    arity,
105                    start_address,
106                },
107            ) => {
108                self.stack_push(arity);
109                self.stack_push(start_address);
110                self.execute_instruction(inst).map(Effects::one)
111            }
112            (inst, CPUMessage::None) => self.execute_instruction(inst).map(Effects::one),
113        }
114    }
115}

117impl CPU {
118    pub fn op_span(&mut self, file: u32, start: u32, end: u32) -> eyre::Result<StepOutcome> {
119        self.span = (file, start, end);
120        Ok(StepOutcome::Continue)
121    }
122
123    pub fn op_branch(&mut self, target: u32) -> eyre::Result<StepOutcome> {
124        self.set_pending_pc(target);
125        Ok(StepOutcome::Continue)
126    }
127
128    pub fn op_conditional_branch(
129        &mut self,
130        true_target: u32,
131        false_target: u32,
132    ) -> eyre::Result<StepOutcome> {
133        let cond = self.stack.pop().ok_or(eyre::eyre!("stack underflow"))?;
134        match cond {
135            Value::Bool(true) => {
136                self.set_pending_pc(true_target);
137                Ok(StepOutcome::Continue)
138            }
139            Value::Bool(false) => {
140                self.set_pending_pc(false_target);
141                Ok(StepOutcome::Continue)
142            }
143            _ => Err(eyre::eyre!("type error: expected bool on stack")),
144        }
145    }
146
147    pub fn op_return(&mut self, keep: u32) -> eyre::Result<StepOutcome> {
148        let frame = self.pop_frame()?;
149        if self.stack.len() - frame.base < (keep as usize) {
150            return Err(eyre::eyre!("not enough values to return"));
151        }
152
153        // Collect return values before truncating
154        let top = self.stack.len() - keep as usize;
155        let return_values: Vec<Value> = self.stack[top..].to_vec();
156        self.stack.drain(frame.base..top);
157
158        if self.get_frame().is_err() {
159            // No more frames - program is returning
160            self.set_return_values(return_values);
161            Ok(StepOutcome::Return)
162        } else {
163            self.set_pending_pc(frame.ret_pc);
164            Ok(StepOutcome::Continue)
165        }
166    }
167
168    pub fn op_call(&mut self, arity: u32, target: u32) -> eyre::Result<StepOutcome> {
169        if self.stack.len() < (arity as usize) {
170            return Err(eyre::eyre!(
171                "not enough arguments on stack to call function"
172            ));
173        }
174
175        let base = self.stack.len() - (arity as usize);
176        let frame = Frame {
177            base,
178            span: self.span,
179            function: None,
180            ret_pc: self.current_pc + 1,
181        };
182        self.push_frame(frame);
183        self.set_pending_pc(target);
184        Ok(StepOutcome::Continue)
185    }
186
187    pub fn op_indirect_call(&mut self) -> eyre::Result<StepOutcome> {
188        // simliar order to op_call but from the stack
189        let target: u32 = self.stack_pop()?.try_into()?;
190        let arity: u32 = self.stack_pop()?.try_into()?;
191        let f = self.stack_pop()?.get_function_ref()?;
192
193        if self.stack.len() < (arity as usize) {
194            return Err(eyre::eyre!(
195                "not enough arguments on stack to call function"
196            ));
197        }
198
199        let base = self.stack.len() - (arity as usize);
200        let frame = Frame {
201            base,
202            span: self.span,
203            function: Some(f as usize),
204            ret_pc: self.current_pc + 1,
205        };
206        self.push_frame(frame);
207        self.set_pending_pc(target);
208        Ok(StepOutcome::Continue)
209    }
210
211    fn op_load(&mut self, ty: Type, addr: u32) -> eyre::Result<StepOutcome> {
212        // addr should be local to frame.
213        let value = self.get_local(addr as usize)?;
214        if value.type_of() != ty {
215            return Err(eyre::eyre!(format!(
216                "type error: expected {:?} at address {}, got {:?}",
217                ty,
218                addr,
219                value.type_of()
220            )));
221        }
222        self.stack_push(*value);
223        Ok(StepOutcome::Continue)
224    }
225
226    pub fn op_store(&mut self, ty: Type, addr: u32) -> Result<StepOutcome> {
227        let v: Value = self.stack_pop()?;
228        log::debug!("store value {:?} at addr {}", v, addr);
229        if !v.is_undefined() && v.type_of() != ty {
230            return Err(eyre::eyre!("Type mismatch"));
231        }
232        *self.get_local_mut(addr as usize)? = v;
233        Ok(StepOutcome::Continue)
234    }
235
236    pub fn op_dup(&mut self) -> Result<StepOutcome> {
237        let v = *self.stack_top()?;
238        self.stack.push(v);
239        Ok(StepOutcome::Continue)
240    }
241
242    pub fn op_heap_alloc(&mut self, n_elements: u32) -> Result<StepOutcome> {
243        let n: usize = n_elements as usize;
244        if self.stack.len() < n {
245            return Err(eyre::eyre!("stack underflow"));
246        }
247        let start = self.stack.len() - n;
248        let values: Box<[Value]> = self.stack.drain(start..).collect();
249        let heap_id = self.push_heap_object(values);
250        self.stack_push(Value::HeapRef(heap_id));
251        Ok(StepOutcome::Continue)
252    }
253
254    pub fn op_get_item(&mut self) -> Result<StepOutcome> {
255        let index = Self::heap_index(self.stack_pop()?)?;
256        let heap_id = self.stack_pop()?.get_heap_ref()?;
257        let value = *self
258            .heap_object(heap_id)?
259            .get(index)
260            .ok_or_else(|| eyre::eyre!("heap index {} out of bounds", index))?;
261        self.stack_push(value);
262        Ok(StepOutcome::Continue)
263    }
264
265    pub fn op_heap_dealloc(&mut self) -> Result<StepOutcome> {
266        let id = self.stack_pop()?.get_heap_ref()?;
267        self.dealloc_heap_object(id)?;
268        Ok(StepOutcome::Continue)
269    }
270
271    pub fn op_const(&mut self, v: Value) -> Result<StepOutcome> {
272        self.stack.push(v);
273        Ok(StepOutcome::Continue)
274    }
275
276    fn heap_index(value: Value) -> Result<usize> {
277        match value {
278            Value::U32(index) => Ok(index as usize),
279            Value::U64(index) => usize::try_from(index)
280                .map_err(|_| eyre::eyre!("heap index {} does not fit in usize", index)),
281            Value::I64(index) if index >= 0 => usize::try_from(index)
282                .map_err(|_| eyre::eyre!("heap index {} does not fit in usize", index)),
283            Value::I64(index) => Err(eyre::eyre!(
284                "heap index must be non-negative, got {}",
285                index
286            )),
287            _ => Err(eyre::eyre!(
288                "type error: expected integer heap index, got {:?}",
289                value.type_of()
290            )),
291        }
292    }
293}

691        pub fn $name(&mut self, ty: Type) -> Result<StepOutcome> {
692            let lhs: Value = self.stack_pop()?;
693            let rhs: Value = self.stack_pop()?;
694            if lhs.type_of() != ty {
695                return Err(eyre::eyre!(
696                    "Type mismatch, expected {} got {} for lhs",
697                    ty,
698                    lhs.type_of()
699                ));
700            }
701
702            if rhs.type_of() != ty {
703                return Err(eyre::eyre!(
704                    "Type mismatch, expected {} got {} for rhs",
705                    ty,
706                    rhs.type_of()
707                ));
708            }
709
710            let output = match (lhs, rhs) {
711                (Value::I64(l), Value::I64(r)) => Value::I64(l.$op(r)),
712                (Value::U32(l), Value::U32(r)) => Value::U32(l.$op(r)),
713                (Value::U64(l), Value::U64(r)) => Value::U64(l.$op(r)),
714                (Value::F64(l), Value::F64(r)) => Value::F64(l.$op(r)),
715                _ => {
716                    return Err(eyre::eyre!(
717                        "cannot {} {} and {}",
718                        stringify!($op),
719                        lhs.type_of(),
720                        rhs.type_of()
721                    ))
722                }
723            };
724            self.stack.push(output);
725            Ok(StepOutcome::Continue)
726        }
727    };
