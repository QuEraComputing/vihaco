1//! Parsed-syntax data shapes. See module docs in [`super`].
2
3/// Parsed `.sst` module — pre-resolution. `H` is the consumer's device-header
4/// enum (typically derives `Parse` via Item 5's `DeviceHeader`).
5#[derive(Debug, Clone, PartialEq)]
6pub struct ParsedModule<I, H> {
7    pub headers: Vec<H>,
8    pub functions: Vec<ParsedFunction<I>>,
9}
10
11#[derive(Debug, Clone, PartialEq)]
12pub struct ParsedFunction<I> {
13    /// Function name with the leading `@` stripped (`@main` → `"main"`).
14    pub name: String,
15    /// Empty for the moment — `.sst` examples don't exercise parameters.
16    /// Non-empty parameter syntax errors during parsing.
17    pub params: Vec<Param>,
18    /// Return type as a bare token (`i64`, `f64`, …). Resolver converts to
19    /// `vihaco::Type`.
20    pub return_ty: Option<RawType>,
21    pub body: Vec<BodyItem<I>>,
22}
23
24#[derive(Debug, Clone, PartialEq)]
25pub struct Param {
26    pub name: String,
27    pub ty: RawType,
28}
29
30/// Bare type token — `"i64"`, `"f64"`, …. The resolver translates to
31/// [`crate::value::Type`].
32#[derive(Debug, Clone, PartialEq, Eq, Hash)]
33pub struct RawType(pub String);
34
35/// One source-level body item. Resolved into zero-or-more `I` values.
36#[derive(Debug, Clone, PartialEq)]
37pub enum BodyItem<I> {
38    /// Canonical, derive-parsed instruction — `I::parser()` succeeded.
39    Direct(I),
40    /// Untyped source form — sugar (`poly <addr> 1.0`) or symbolic operand
41    /// (`br @label`, `const.str "hi"`) that needs interner / symbol-table
42    /// access. The consumer's [`super::Resolve`] impl expands it.
43    Raw(RawForm),
44}
45
46/// Untyped source form: a mnemonic followed by some operands. The shape is
47/// intentionally lossless — the resolver decides what each form means based
48/// on the mnemonic.
49#[derive(Debug, Clone, PartialEq)]
50pub struct RawForm {
51    pub mnemonic: String,
52    pub operands: Vec<RawOperand>,
53}
54
55#[derive(Debug, Clone, PartialEq)]
56pub enum RawOperand {
57    /// Bare token: `AOD0:T1:A`, `DIGI:0`, `i64`, dotted device names.
58    Ident(String),
59    Int(i64),
60    UInt(u64),
61    Float(f64),
62    Bool(bool),
63    /// Contents of `"…"` with escape sequences decoded.
64    StringLit(String),
65    /// `@name` form — the `@` is consumed; the name remains.
66    Symbol(String),
67}
