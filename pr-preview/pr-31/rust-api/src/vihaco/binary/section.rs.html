13/// The fully resolved path for a given section.
14///
15/// The root section will be empty.
16#[derive(Clone, PartialEq, Eq, PartialOrd, Ord, Hash)]
17pub struct SectionPath {
18    components: Vec<String>,
19}
20
21impl SectionPath {
22    pub fn root() -> Self {
23        Self {
24            components: Vec::new(),
25        }
26    }
27
28    pub fn is_root(&self) -> bool {
29        self.components.is_empty()
30    }
31
32    pub fn components(&self) -> &[String] {
33        &self.components
34    }
35
36    pub fn local_name(&self) -> Option<&str> {
37        self.components.last().map(String::as_str)
38    }
39
40    pub fn child(&self, local_name: impl Into<String>) -> Self {
41        let mut components = self.components.clone();
42        components.push(local_name.into());
43        Self { components }
44    }
45}
46
47impl Default for SectionPath {
48    fn default() -> Self {
49        Self::root()
50    }
51}
52
53impl std::fmt::Display for SectionPath {
54    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
55        if self.is_root() {
56            return f.write_str("<root>");
57        }
58
59        for (index, component) in self.components.iter().enumerate() {
60            if index != 0 {
61                f.write_str("/")?;
62            }
63            f.write_str(component)?;
64        }
65        Ok(())
66    }
67}
68
69impl std::fmt::Debug for SectionPath {
70    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
71        std::fmt::Display::fmt(self, f)
72    }
73}

77/// For the public handle of a section, see [`SectionView`].

81    pub section: Range<usize>,
82    pub header: Range<usize>,
83    pub bytecode: Range<usize>,
84    pub children: Vec<SectionNode>,

87/// The public handle of a bytecode section.
88///
89/// This is a lightweight view into information owned by [`BytecodeFile`].
90pub struct SectionView<'bc, F = Vec<u8>, C = ProgramContext>
91where
92    F: FileContents,
93    C: BytecodeContext,
94{
95    pub(super) contents: &'bc F,
96    pub(super) context: ContextHandle<C>,
97    pub(super) node: &'bc SectionNode,
98}
99
100impl<'bc, F, C> Clone for SectionView<'bc, F, C>
101where
102    F: FileContents,
103    C: BytecodeContext,
104{
105    fn clone(&self) -> SectionView<'bc, F, C> {
106        SectionView {
107            contents: self.contents,
108            context: self.context.clone(),
109            node: self.node,
110        }
111    }
112}
113
114impl<'bc, F, C> SectionView<'bc, F, C>
115where
116    F: FileContents,
117    C: BytecodeContext,
118{
119    pub fn path(&self) -> &'bc SectionPath {
120        &self.node.path
121    }
122
123    pub fn display_path(&self) -> &'bc SectionPath {
124        &self.node.path
125    }
126
127    pub fn context(&self) -> &C {
128        self.context.get()
129    }
130
131    pub fn context_handle(&self) -> ContextHandle<C> {
132        self.context.clone()
133    }
134
135    pub fn children(&self) -> impl Iterator<Item = SectionView<'bc, F, C>> + '_ {
136        self.node.children.iter().map(|node| SectionView {
137            contents: self.contents,
138            context: self.context.clone(),
139            node,
140        })
141    }
142
143    pub fn child(&self, local_name: &str) -> Option<SectionView<'bc, F, C>> {
144        self.node
145            .children
146            .iter()
147            .find(|child| {
148                child
149                    .path
150                    .local_name()
151                    .is_some_and(|name| name == local_name)
152            })
153            .map(|node| SectionView {
154                contents: self.contents,
155                context: self.context.clone(),
156                node,
157            })
158    }
159
160    pub fn local_name(&self) -> Option<&str> {
161        self.node.path.local_name()
162    }
163}

165pub type BinarySectionView<'bc, C> = SectionView<'bc, Vec<u8>, C>;
166
167impl<'bc, C> BinarySectionView<'bc, C>
168where
169    C: BytecodeContext,
170{
171    pub fn header_bytes(&self) -> &'bc [u8] {
172        &self.contents[self.node.header.clone()]
173    }
174
175    pub fn bytecode(&self) -> &'bc [u8] {
176        &self.contents[self.node.bytecode.clone()]
177    }
178
179    /// Parse the specified composite header from the raw header bytes.
180    pub fn parse_header<H: CompositeHeader>(&self) -> eyre::Result<H> {
181        let bytes = self.header_bytes();
182        let mut cursor = Cursor::new(bytes);
183        let header = H::from_bytes(&mut cursor)?;
184        if cursor.position() as usize != bytes.len() {
185            return Err(eyre::eyre!(
186                "section `{}` header has {} trailing bytes",
187                self.display_path(),
188                bytes.len() - cursor.position() as usize
189            ));
190        }
191        Ok(header)
192    }
193}

195impl<C> std::fmt::Debug for BinarySectionView<'_, C>
196where
197    C: BytecodeContext,
198{
199    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
200        f.debug_struct("SectionView")
201            .field("path", &self.display_path().to_string())
202            .field("header_len", &self.header_bytes().len())
203            .field("bytecode_len", &self.bytecode().len())
204            .field("child_count", &self.node.children.len())
205            .finish()
206    }
207}
