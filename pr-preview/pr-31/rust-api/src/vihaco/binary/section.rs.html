14/// The fully resolved path for a given section.
15///
16/// The root section will be empty.
17#[derive(Clone, PartialEq, Eq, PartialOrd, Ord, Hash)]
18pub struct SectionPath {
19    components: Vec<String>,
20}
21
22impl SectionPath {
23    pub fn root() -> Self {
24        Self {
25            components: Vec::new(),
26        }
27    }
28
29    pub fn is_root(&self) -> bool {
30        self.components.is_empty()
31    }
32
33    pub fn components(&self) -> &[String] {
34        &self.components
35    }
36
37    pub fn local_name(&self) -> Option<&str> {
38        self.components.last().map(String::as_str)
39    }
40
41    pub fn child(&self, local_name: impl Into<String>) -> Self {
42        let mut components = self.components.clone();
43        components.push(local_name.into());
44        Self { components }
45    }
46}
47
48impl Default for SectionPath {
49    fn default() -> Self {
50        Self::root()
51    }
52}
53
54impl std::fmt::Display for SectionPath {
55    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
56        if self.is_root() {
57            return f.write_str("<root>");
58        }
59
60        for (index, component) in self.components.iter().enumerate() {
61            if index != 0 {
62                f.write_str("/")?;
63            }
64            f.write_str(component)?;
65        }
66        Ok(())
67    }
68}
69
70impl std::fmt::Debug for SectionPath {
71    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
72        std::fmt::Display::fmt(self, f)
73    }
74}

78/// For the public handle of a section, see [`SectionView`].

80pub(super) struct SectionNode {
81    pub path: SectionPath,
82    pub section: Range<usize>,
83    pub header: Range<usize>,
84    pub bytecode: Range<usize>,
85    pub children: Vec<SectionNode>,
86}

88/// The public handle of a bytecode section.
89///
90/// This is a lightweight view into information owned by [`BytecodeFile`].
91pub struct SectionView<'bc, F = Vec<u8>, C = ProgramContext>
92where
93    F: FileContents,
94    C: BytecodeContext,
95{
96    pub(super) contents: &'bc F,
97    pub(super) context: ContextHandle<C>,
98    pub(super) node: &'bc SectionNode,
99}
100
101impl<'bc, F, C> Clone for SectionView<'bc, F, C>
102where
103    F: FileContents,
104    C: BytecodeContext,
105{
106    fn clone(&self) -> SectionView<'bc, F, C> {
107        SectionView {
108            contents: self.contents,
109            context: self.context.clone(),
110            node: self.node,
111        }
112    }
113}
114
115impl<'bc, F, C> SectionView<'bc, F, C>
116where
117    F: FileContents,
118    C: BytecodeContext,
119{
120    pub fn path(&self) -> &'bc SectionPath {
121        &self.node.path
122    }
123
124    pub fn display_path(&self) -> &'bc SectionPath {
125        &self.node.path
126    }
127
128    pub fn context_handle(&self) -> ContextHandle<C> {
129        self.context.clone()
130    }
131
132    pub fn children(&self) -> impl Iterator<Item = SectionView<'bc, F, C>> + '_ {
133        self.node.children.iter().map(|node| SectionView {
134            contents: self.contents,
135            context: self.context.clone(),
136            node,
137        })
138    }
139
140    pub fn child(&self, local_name: &str) -> Option<SectionView<'bc, F, C>> {
141        self.node
142            .children
143            .iter()
144            .find(|child| {
145                child
146                    .path
147                    .local_name()
148                    .is_some_and(|name| name == local_name)
149            })
150            .map(|node| SectionView {
151                contents: self.contents,
152                context: self.context.clone(),
153                node,
154            })
155    }
156
157    pub fn local_name(&self) -> Option<&str> {
158        self.node.path.local_name()
159    }
160}

162pub type BinarySectionView<'bc, C> = SectionView<'bc, Vec<u8>, C>;
163
164impl<'bc, C> BinarySectionView<'bc, C>
165where
166    C: BytecodeContext,
167{
168    pub fn header_bytes(&self) -> &'bc [u8] {
169        &self.contents[self.node.header.clone()]
170    }
171
172    pub fn bytecode(&self) -> &'bc [u8] {
173        &self.contents[self.node.bytecode.clone()]
174    }
175
176    /// Parse the specified composite header from the raw header bytes.
177    pub fn parse_header<H: CompositeHeader>(&self) -> eyre::Result<H> {
178        let bytes = self.header_bytes();
179        let mut cursor = Cursor::new(bytes);
180        let header = H::from_bytes(&mut cursor)?;
181        if cursor.position() as usize != bytes.len() {
182            return Err(eyre::eyre!(
183                "section `{}` header has {} trailing bytes",
184                self.display_path(),
185                bytes.len() - cursor.position() as usize
186            ));
187        }
188        Ok(header)
189    }
190}
191
192impl<C> std::fmt::Debug for BinarySectionView<'_, C>
193where
194    C: BytecodeContext,
195{
196    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
197        f.debug_struct("SectionView")
198            .field("path", &self.display_path().to_string())
199            .field("header_len", &self.header_bytes().len())
200            .field("bytecode_len", &self.bytecode().len())
201            .field("child_count", &self.node.children.len())
202            .finish()
203    }
204}

206pub type TextSectionView<'bc, C> = SectionView<'bc, String, C>;
207
208impl<'bc, C> TextSectionView<'bc, C>
209where
210    C: BytecodeContext,
211{
212    pub fn header_text(&self) -> &'bc str {
213        &self.contents[self.node.header.clone()]
214    }
215
216    pub fn text(&self) -> &'bc str {
217        &self.contents[self.node.bytecode.clone()]
218    }
219
220    /// Parse the specified composite header from the text format.
221    pub fn parse_header<H: CompositeHeader>(&self) -> eyre::Result<H> {
222        let text = self.header_text();
223        let mut cursor = Cursor::new(text);
224        let header = H::from_text(&mut cursor)?;
225        if cursor.position() as usize != text.len() {
226            return Err(eyre::eyre!(
227                "section `{}` header has {} trailing bytes",
228                self.display_path(),
229                text.len() - cursor.position() as usize
230            ));
231        }
232        Ok(header)
233    }
234}
