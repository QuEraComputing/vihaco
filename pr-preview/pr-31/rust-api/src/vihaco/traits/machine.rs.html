10pub trait ProgramCounter {
11    type Instruction: Instruction;
12
13    /// Get the current program counter.
14    fn pc(&self) -> u32;
15
16    /// Get a mutable reference to the program counter.
17    fn pc_mut(&mut self) -> &mut u32;
18
19    /// Get the instruction at the given program counter without advancing it.
20    fn get_instruction(&self, pc: u32) -> Result<&Self::Instruction>;
21
22    /// Peek the current instruction without advancing the program counter.
23    fn peek_instruction(&self) -> Result<&Self::Instruction> {
24        let pc = self.pc();
25        self.get_instruction(pc)
26    }
27
28    /// Get the current instruction and advance the program counter by 1.
29    fn next_instruction(&mut self) -> Result<&Self::Instruction> {
30        let pc = self.pc();
31        if let Some(v) = self.pc_mut().checked_add(1) {
32            *self.pc_mut() = v;
33        }
34        self.get_instruction(pc)
35    }
36}

38pub trait StackMemory {
39    type Value;
40    fn stack(&self) -> &Vec<Self::Value>;
41    fn stack_mut(&mut self) -> &mut Vec<Self::Value>;
42    fn stack_push<T: Into<Self::Value>>(&mut self, v: T);
43    fn stack_pop(&mut self) -> Result<Self::Value>;
44    fn stack_get(&self, pos: usize) -> Result<&Self::Value>;
45    fn stack_get_mut(&mut self, pos: usize) -> Result<&mut Self::Value>;
46    fn stack_len(&self) -> usize;
47    fn stack_is_empty(&self) -> bool {
48        self.stack_len() == 0
49    }
50    fn stack_top(&self) -> Result<&Self::Value> {
51        if self.stack_is_empty() {
52            Err(eyre::eyre!("stack underflow"))
53        } else {
54            self.stack_get(self.stack_len() - 1)
55        }
56    }
57    fn stack_top_mut(&mut self) -> Result<&mut Self::Value> {
58        let len = self.stack_len();
59        if len == 0 {
60            Err(eyre::eyre!("stack underflow"))
61        } else {
62            self.stack_get_mut(len - 1)
63        }
64    }
65}

67pub trait StackFrame {
68    fn get_frame(&self) -> Result<&Frame>;
69    fn get_frame_mut(&mut self) -> Result<&mut Frame>;
70    fn push_frame(&mut self, frame: Frame);
71    fn pop_frame(&mut self) -> Result<Frame>;
72}

74pub trait FrameMemory: StackFrame + StackMemory {
75    fn frame_base(&self) -> Result<usize> {
76        self.get_frame().map(|f| f.base)
77    }
78
79    fn get_local(&self, index: usize) -> Result<&Self::Value> {
80        let base = self.frame_base()?;
81        self.stack_get(base + index)
82    }
83
84    fn get_local_mut(&mut self, index: usize) -> Result<&mut Self::Value> {
85        let base = self.frame_base()?;
86        self.stack_get_mut(base + index)
87    }
88}
