1//! The `Resolve` trait — bridge between [`super::ParsedModule`] and the
2//! runtime [`crate::module::Module`]. Each consumer implements this for its
3//! own instruction set + header type, holding whatever state is needed
4//! (label table, string interner, sugar expansion rules).
5
6use crate::syntax::{BodyItem, ParsedModule};
7
8/// Lower a parsed module to its resolved runtime form.
9///
10/// Implementations own the symbol table / string interner / sugar expansion
11/// rules. Default behavior for any `BodyItem::Raw` is to error; consumers
12/// override [`Resolve::resolve_body`] to handle their sugar/symbolic forms.
13pub trait Resolve<I, H> {
14    /// Resolved module type — concrete to the consumer (typically
15    /// `crate::module::Module<I, Value, Type, Info>` with consumer-specific
16    /// `Info`).
17    type Module;
18
19    fn resolve_module(&mut self, parsed: ParsedModule<I, H>) -> eyre::Result<Self::Module>;
20
21    /// Convenience hook: lower a single function body. Default just walks
22    /// `Direct` items and errors on `Raw`. Override to handle sugar/symbolic
23    /// forms.
24    fn resolve_body(&mut self, items: Vec<BodyItem<I>>) -> eyre::Result<Vec<I>> {
25        items
26            .into_iter()
27            .map(|item| match item {
28                BodyItem::Direct(inst) => Ok(inst),
29                BodyItem::Raw(raw) => Err(eyre::eyre!(
30                    "unhandled raw form `{}` (consumer's Resolve impl must override `resolve_body`)",
31                    raw.mnemonic
32                )),
33            })
34            .collect()
35    }
36}
