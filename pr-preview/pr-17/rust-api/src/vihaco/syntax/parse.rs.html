1//! chumsky-0.10 combinators for the parsed-syntax shape.
2//!
3//! `Parse` impls for `ParsedModule`/`ParsedFunction` are generic over the
4//! consumer's instruction type `I` and device-header type `H`. The body
5//! parser tries `I::parser()` first (with `.rewind()` so failure restores
6//! input) and falls back to [`raw_form`] for everything `I` can't accept.
7
8use chumsky::error::Simple;
9use chumsky::extra;
10use chumsky::prelude::*;
11use vihaco_parser_core::Parse;

13use crate::syntax::{BodyItem, Param, ParsedFunction, ParsedModule, RawForm, RawOperand, RawType};

15type E<'src> = extra::Err<Simple<'src, char>>;

17/// Whitespace and `//`-to-end-of-line comments. Zero-or-more.
18pub fn skip<'src>() -> impl Parser<'src, &'src str, (), E<'src>> + Clone {
19    let ws = any().filter(|c: &char| c.is_whitespace()).ignored();
20    let line_comment = just("//")
21        .then(any().and_is(just('\n').not()).repeated())
22        .ignored();
23    ws.or(line_comment).repeated().collect::<()>()
24}

26/// `"…"` with `\\`, `\"`, `\n`, `\t`, `\r` escape sequences.
27pub fn string_literal<'src>() -> impl Parser<'src, &'src str, String, E<'src>> + Clone {
28    let escape = just('\\').ignore_then(choice((
29        just('"').to('"'),
30        just('\\').to('\\'),
31        just('n').to('\n'),
32        just('t').to('\t'),
33        just('r').to('\r'),
34        just('0').to('\0'),
35    )));
36    let char_or_escape = choice((
37        escape,
38        any().and_is(just('"').not()).and_is(just('\\').not()),
39    ));
40    just('"')
41        .ignore_then(char_or_escape.repeated().collect::<String>())
42        .then_ignore(just('"'))
43}

45/// `@name` — leading `@` consumed; name is `ident()`-shaped.
46pub fn symbol_ref<'src>() -> impl Parser<'src, &'src str, String, E<'src>> + Clone {
47    just('@').ignore_then(vihaco_parser_core::ident())
48}

50/// One operand: tries each shape in order. `Ident` is the catch-all and runs
51/// last so numeric literals get their typed shape.
52pub fn raw_operand<'src>() -> impl Parser<'src, &'src str, RawOperand, E<'src>> + Clone {
53    // Float: int `.` digits, optional `e[+-]?digits` for scientific notation.
54    // Required `.` distinguishes from ints; ints would otherwise match the
55    // leading-integer prefix.
56    let exp = || {
57        one_of("eE")
58            .then(one_of("+-").or_not())
59            .then(text::digits(10))
60    };
61    let float_lit = text::int(10)
62        .then(just('.').then(text::digits(10)))
63        .then(exp().or_not())
64        .to_slice()
65        .map(|s: &str| RawOperand::Float(s.parse().unwrap()));
66    let neg_float = just('-')
67        .then(text::int(10))
68        .then(just('.').then(text::digits(10)))
69        .then(exp().or_not())
70        .to_slice()
71        .map(|s: &str| RawOperand::Float(s.parse().unwrap()));
72    let neg_int = just('-')
73        .then(text::int(10))
74        .to_slice()
75        .map(|s: &str| RawOperand::Int(s.parse().unwrap()));
76    let uint_lit = text::int(10).map(|s: &str| RawOperand::UInt(s.parse().unwrap()));
77    let bool_lit = choice((
78        just("true").to(RawOperand::Bool(true)),
79        just("false").to(RawOperand::Bool(false)),
80    ));
81    let str_lit = string_literal().map(RawOperand::StringLit);
82    let sym = symbol_ref().map(RawOperand::Symbol);
83    let id = vihaco_parser_core::ident().map(RawOperand::Ident);
84
85    choice((
86        str_lit, sym, bool_lit, neg_float, float_lit, neg_int, uint_lit, id,
87    ))
88}

90/// `mnemonic operand (, operand)* ` — one source line's worth, sans
91/// terminator.
92pub fn raw_form<'src>() -> impl Parser<'src, &'src str, RawForm, E<'src>> + Clone {
93    let inline_ws = any()
94        .filter(|c: &char| c.is_whitespace() && *c != '\n')
95        .repeated();
96    let operand_sep = inline_ws.then(just(',').or_not()).then(inline_ws);
97    vihaco_parser_core::ident()
98        .then(
99            inline_ws
100                .ignore_then(raw_operand())
101                .separated_by(operand_sep.ignored())
102                .collect::<Vec<_>>(),
103        )
104        .map(|(mnemonic, operands)| RawForm { mnemonic, operands })
105}

107/// `RawType` is a bare identifier — `i64`, `f64`, …. Resolver translates.

109    vihaco_parser_core::ident().map(RawType)

112/// Block-body helper: parse a sequence of whitespace-separated `i64`s as
113/// **flat rows** between an outer `{` … `}` provided by the caller (typically
114/// the `#[derive(Parse)]`-emitted delimiters). The body itself is a sequence
115/// of `i64`s with any whitespace (including newlines) between them.
116///
117/// Real usage: `device slm.filling { 0 1 2 3 };`.
118pub fn block_i64_flat<'src>() -> impl Parser<'src, &'src str, Vec<i64>, E<'src>> + Clone {
119    let item = just('-')
120        .or_not()
121        .then(text::int(10))
122        .to_slice()
123        .map(|s: &str| s.parse::<i64>().unwrap());
124    // Allow any whitespace (incl. newlines) between numbers and around them.
125    let ws = any().filter(|c: &char| c.is_whitespace()).repeated();
126    ws.ignore_then(item)
127        .repeated()
128        .collect::<Vec<_>>()
129        .then_ignore(ws)
130}

132/// Block-body helper: parse rows of `i64 i64` pairs separated by whitespace.
133/// Within a row the two ints are whitespace-separated; rows themselves are
134/// also just whitespace-separated (newlines or other ws — the helper doesn't
135/// require row alignment).
136///
137/// Real usage: `device slm.traps { 1 1\n 5 1\n ... };`,
138/// `device camera.detect_sites { ... };`.
139pub fn block_i64_pairs<'src>() -> impl Parser<'src, &'src str, Vec<(i64, i64)>, E<'src>> + Clone {
140    let signed_int = just('-')
141        .or_not()
142        .then(text::int(10))
143        .to_slice()
144        .map(|s: &str| s.parse::<i64>().unwrap());
145    let inline_ws = any()
146        .filter(|c: &char| c.is_whitespace() && *c != '\n')
147        .repeated()
148        .at_least(1);
149    let ws = any().filter(|c: &char| c.is_whitespace()).repeated();
150
151    let pair = signed_int.then_ignore(inline_ws).then(signed_int);
152    ws.ignore_then(pair)
153        .repeated()
154        .collect::<Vec<_>>()
155        .then_ignore(ws)
156}

158/// Lookahead-only end-of-statement marker. A body item ends at a newline,
159/// `}`, or end-of-input — anything else means the canonical parse consumed
160/// only a prefix of a longer source line (e.g. matching the unit variant
161/// `fpga::Play` on `fpga::Play 5`, which is actually the sugar form).
162///
163/// `.rewind()` so we don't consume the terminator — the outer body parser's
164/// `skip()` handles whitespace/newlines between items.
165fn statement_end<'src>() -> impl Parser<'src, &'src str, (), E<'src>> + Clone {
166    let inline_ws = any()
167        .filter(|c: &char| c.is_whitespace() && *c != '\n')
168        .repeated();
169    let terminator = choice((just('\n').ignored(), just('}').ignored(), end())).rewind();
170    inline_ws.ignore_then(terminator)
171}

173/// Body item: try `I::parser()` first, but only accept it if the remainder of
174/// the line is empty (canonical form must own the whole statement). On
175/// failure, fall back to `raw_form()` which captures sugar and symbolic
176/// operand forms.
177pub fn body_item<'src, I>() -> impl Parser<'src, &'src str, BodyItem<I>, E<'src>>
178where
179    I: Parse<'src> + 'src,
180{
181    let direct = I::parser()
182        .then_ignore(statement_end())
183        .map(BodyItem::Direct);
184    let raw = raw_form().map(BodyItem::Raw);
185    direct.boxed().or(raw.boxed())
186}

188/// Parse `i64`/`f64`/etc. parameter list. Currently only accepts empty `()`.
189fn param_list<'src>() -> impl Parser<'src, &'src str, Vec<Param>, E<'src>> + Clone {
190    just('(').padded().then(just(')').padded()).to(Vec::new())
191}

193impl<'src, I> Parse<'src> for ParsedFunction<I>
194where
195    I: Parse<'src> + 'src,
196{
197    fn parser() -> impl Parser<'src, &'src str, Self, E<'src>> {
198        let return_ty = just("->").padded().ignore_then(raw_type()).or_not();
199        let body = skip()
200            .ignore_then(body_item::<I>())
201            .repeated()
202            .collect::<Vec<_>>()
203            .then_ignore(skip());
204
205        just("fn")
206            .padded()
207            .ignore_then(just('@'))
208            .ignore_then(vihaco_parser_core::ident())
209            .then(param_list())
210            .then(return_ty)
211            .then_ignore(just('{').padded())
212            .then(body)
213            .then_ignore(just('}').padded())
214            .map(|(((name, params), return_ty), body)| ParsedFunction {
215                name,
216                params,
217                return_ty,
218                body,
219            })
220    }
221}

223impl<'src, I, H> Parse<'src> for ParsedModule<I, H>
224where
225    I: Parse<'src> + 'src,
226    H: Parse<'src> + 'src,
227{
228    fn parser() -> impl Parser<'src, &'src str, Self, E<'src>> {
229        // Each header line: `H::parser()` terminated by `;` and surrounded by
230        // skippable whitespace/comments.
231        let header_line = H::parser().then_ignore(just(';').padded());
232
233        let headers = skip()
234            .ignore_then(header_line)
235            .repeated()
236            .collect::<Vec<_>>();
237
238        let functions = skip()
239            .ignore_then(ParsedFunction::<I>::parser())
240            .repeated()
241            .collect::<Vec<_>>()
242            .then_ignore(skip());
243
244        headers
245            .then(functions)
246            .map(|(headers, functions)| ParsedModule { headers, functions })
247    }
248}
