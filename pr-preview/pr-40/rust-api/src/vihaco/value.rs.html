6use crate::{

11/// A value for a `vihaco` component.
12///
13/// Values must be associated with some [`Type`], connected by both
14/// [`Value::Type`] and [`Value::type_of`].
15pub trait Value: Sized + ByteWidth + FromBytes + WriteBytes {
16    /// The type universe that this value set is contained within.
17    type Type: Type;
18
19    fn type_of(&self) -> Self::Type;
20}
21
22/// The type universe of a `vihaco` component.
23///
24/// See [`Value`] for the value set of a `vihaco` component.
25pub trait Type: Sized + ByteWidth + FromBytes + WriteBytes {}
26
27/// A marker trait to denote a type is a resolution context to be used
28/// during the resolution of a [`crate::syntax::ParsedModule`] into a
29/// [`crate::module::LocalModule`].
30pub trait ResolutionContext {}
31
32/// The syntactic representation of a value.
33///
34/// This is used in raw forms during module parsing.
35///
36/// As the author of a component, you should implement this for its
37/// instruction set. The author of the architecture using the component
38/// can then have its own [`ResolutionContext`] to describe
39/// architecture-specific implementation of dialect requirements.
40pub trait SyntacticValue<'src, Ctx: ResolutionContext>: Parse<'src> {
41    type RuntimeForm: Value;
42
43    /// Resolve a syntactic value into its runtime form.
44    ///
45    /// `ctx` provides any functionality or mutable state that is
46    /// required for a syntactic value to be resolved. For example,
47    /// `ctx` can provide a function for string interning if your
48    /// component requires it.
49    fn resolve(&self, ctx: &mut Ctx) -> eyre::Result<Self::RuntimeForm>;
50}
